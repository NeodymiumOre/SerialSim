33void MainWindow::readFromDevice()
34{
35 while(this->device->canReadLine())
36 {
37 int init, work, butt;
38 float Ax, Ay, Gx, Gy;
39 bool is_data_ok = false;
40
41 QString line = this->device->readLine();
42
43 // reading control values on fixed positions
44 init = line.at(0).digitValue();
45 work = line.at(2).digitValue();
46 butt = line.at(4).digitValue();
47
48 if(init && work && butt)
49 is_data_ok = true;
50
51 // getting sensor values
52 if(is_data_ok)
53 {
54 // setting number of chars to read
55 //always x.xx = 4 + 1 for sign
56 //(if value is positive it reads " " and cuts it during conversion to float)
57 int char_num = 5;
58 // getting 5 chars from 5th index
59 Ax = line.mid(5, char_num).toFloat();
60 // getting index of next " " after 6th pos
61 int ay_pos = line.indexOf(" ", 6) + 1;
62 // getting 5 chars from next " " char
63 Ay = line.mid(ay_pos, 5).toFloat();
64 // getting index of next " " after ay_pos
65 int gx_pos = line.indexOf(" ", ay_pos) + 1;
66 // getting 5 chars from next " " char
67 Gx = line.mid(gx_pos, 5).toFloat();
68 // getting index of next " " after gx_pos
69 int gy_pos = line.indexOf(" ", gx_pos) + 1;
70 // getting 5 chars from next " " char
71 Gy = line.mid(gy_pos, 5).toFloat();
72
73 qDebug() << Ax << Ay << Gx << Gy;
74 }
75 else
76 {
77 qDebug() << "Broken";
78 }
79
80 }
81}

100void MainWindow::on_pushButtonConnect_clicked()
101{
102 if(ui->comboBoxDevices->count() == 0)
103 {
104 this->addToLogs("Nie wykryto żadnych urządzeń!");
105 return;
106 }
107 if(this->device->isOpen())
108 {
109 this->addToLogs("Port jest już otwarty!");
110 return;
111 }
112
113 QString portName = "/dev/" + ui->comboBoxDevices->currentText();
114 this->device->setPortName(portName);
115 //qDebug() << portName;
116
117 if(device->open(QSerialPort::ReadWrite))
118 {
119 this->device->setBaudRate(QSerialPort::Baud115200);
120 this->device->setParity(QSerialPort::NoParity);
121 this->device->setDataBits(QSerialPort::Data8);
122 this->device->setStopBits(QSerialPort::OneStop);
123 this->device->setFlowControl(QSerialPort::NoFlowControl);
124
125 this->addToLogs("Połączono z urządzeniem " + portName);
126 connect(this->device, SIGNAL(readyRead()), this, SLOT(readFromDevice()));
127 }
128 else
129 {
130 this->addToLogs("Otwarcie portu szeregowego się nie powiodło!");
131 this->addToLogs(device->errorString());
132 qDebug() << this->device->error();
133 }
134}
